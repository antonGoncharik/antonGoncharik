```go
package main

import "fmt"

type Developer struct {
	Name       string
	Focus      string
	Languages  []string
	Frontend   []string
	Backend    []string
	Systems    []string
	Data       []string
	Infra      []string
	Tools      []string
	Environment []string
}

func main() {
	anton := Developer{
		Name:  "Anton",
		Focus: "Backend (Go) with strong Frontend experience",

		Languages: []string{
			"Go", "TypeScript", "JavaScript",
		},

		Frontend: []string{
			"React", "Vue", "Angular",
		},

		Backend: []string{
			"REST", "gRPC", "Microservices", "RabbitMQ",
		},

		Systems: []string{
			"High-load systems",
			"Distributed systems",
			"Performance optimization",
			"Concurrency & memory model",
		},

		Data: []string{
			"PostgreSQL", "Redis",
		},

		Infra: []string{
			"Docker", "CI/CD", "Linux", "Nginx",
		},

		Tools: []string{
			"Git", "Postman", "DBeaver", "curl", "htop",
		},

		Environment: []string{
			"Neovim", "VS Code", "Linux",
		},
	}

	fmt.Printf("👋 Hi, I'm %s\n", anton.Name)
	fmt.Printf("💡 %s\n\n", anton.Focus)

	fmt.Printf("🚀 Languages: %v\n", anton.Languages)
	fmt.Printf("🎨 Frontend:  %v\n", anton.Frontend)
	fmt.Printf("⚙️ Backend:   %v\n\n", anton.Backend)

	fmt.Println("🧠 Systems:")
	for _, s := range anton.Systems {
		fmt.Printf("  • %s\n", s)
	}

	fmt.Printf("\n🗄️ Data: %v\n", anton.Data)
	fmt.Printf("🧱 Infra: %v\n", anton.Infra)

	fmt.Println("\n🛠️ Tools:")
	for _, t := range anton.Tools {
		fmt.Printf("  • %s\n", t)
	}

	fmt.Printf("\n💻 Environment: %v\n", anton.Environment)
}
```
